    class VisionTransformer(nn.Module):
    """ ËßÜËßâ TransformerÔºåÊîØÊåÅË°•‰∏ÅÊàñÊ∑∑Âêà CNN ËæìÂÖ•Èò∂ÊÆµ
    """
    def __init__(self, img_size=224, patch_size=16, in_chans=3, num_classes=1000, embed_dim=768, depth=12,
                 num_heads=12, mlp_ratio=4., qkv_bias=False, mlp_head=False, drop_rate=0., attn_drop_rate=0.):
        super().__init__()
        self.num_classes = num_classes  # Á±ªÂà´Êï∞
        self.num_features = self.embed_dim = embed_dim  # ÁâπÂæÅÊï∞Ôºå‰∏éÂµåÂÖ•Áª¥Â∫¶Áõ∏ÂêåÔºåÁî®‰∫é‰∏éÂÖ∂‰ªñÊ®°Âûã‰øùÊåÅ‰∏ÄËá¥

        # ÂàùÂßãÂåñË°•‰∏ÅÂµåÂÖ•Ê®°Âùó
        self.patch_embed = PatchEmbed(
                img_size=img_size, patch_size=patch_size, in_chans=in_chans, embed_dim=embed_dim)
        num_patches = self.patch_embed.num_patches

        # ÂàùÂßãÂåñ‰ΩçÁΩÆÂµåÂÖ•ÂíåÁ±ªÊ†áËÆ∞
        self.pos_embed = nn.Parameter(torch.zeros(1, num_patches + 1, embed_dim))
        self.cls_token = nn.Parameter(torch.zeros(1, 1, embed_dim))

        # ÂàùÂßãÂåñÂ§ö‰∏™ Transformer ÁºñÁ†ÅÂùó
        self.blocks = nn.ModuleList([
            Block(
                dim=embed_dim, num_heads=num_heads, mlp_ratio=mlp_ratio, qkv_bias=qkv_bias,
                drop=drop_rate, attn_drop=attn_drop_rate)
            for i in range(depth)])

        # ÊúÄÂêéÁöÑÂ±ÇÂΩí‰∏ÄÂåñ
        self.norm = LayerNorm(embed_dim)
        # Ê†πÊçÆÂèÇÊï∞ÈÄâÊã©‰ΩøÁî® MLP ÊàñÂçï‰∏™Á∫øÊÄßÂ±Ç‰Ωú‰∏∫ÂàÜÁ±ªÂ§¥
        if mlp_head:
            self.head = Mlp(embed_dim, int(embed_dim * mlp_ratio), num_classes)
        else:
            self.head = Linear(embed_dim, num_classes)

        # ÂàùÂßãÂåñÊùÉÈáç
        trunc_normal_(self.pos_embed, std=.02)
        trunc_normal_(self.cls_token, std=.02)
        self.apply(self._init_weights)

        # ËæÖÂä©Êìç‰Ωú
        self.pool = IndexSelect()
        self.add = Add()

    def forward(self, x):
        B = x.shape[0]
        x = self.patch_embed(x)

        # Êâ©Â±ïÁ±ªÊ†áËÆ∞ÂíåÂêàÂπ∂Á±ªÊ†áËÆ∞ÂíåË°•‰∏ÅÂµåÂÖ•

        x = torch.cat((cls_tokens, x), dim=1)
        x = self.add([x, self.pos_embed])

        # Â∫îÁî® Transformer ÁºñÁ†ÅÂùó
        for blk in self.blocks:
            x = blk(x)

        # Â∫îÁî®Â±ÇÂΩí‰∏ÄÂåñÂíåÊ±†ÂåñÊìç‰ΩúÔºåÁÑ∂ÂêéÈÄöËøáÂàÜÁ±ªÂ§¥ËæìÂá∫ÊúÄÁªàÁªìÊûú
        x = self.norm(x)
        x = self.pool(x, dim=1, indices=torch.tensor(0, device=x.device))
        x = x.squeeze(1)
        x = self.head(x)
        return x

    def relprop(self, cam=None, method="transformer_attribution", is_ablation=False, start_layer=0, **kwargs):
    # Âü∫Á°ÄÁöÑÁõ∏ÂÖ≥ÊÄß‰º†Êí≠Ôºå‰ªéÂàÜÁ±ªÂ§¥ÂºÄÂßãÂèçÂêë‰º†Êí≠

    elif method == "transformer_attribution" or method == "grad":
        # ‰ΩøÁî® Transformer ÁºñÁ†ÅÂùóÁöÑÊ¢ØÂ∫¶ÂíåÊ≥®ÊÑèÂäõÊùÉÈáçËÆ°ÁÆóÊúÄÁªàÁöÑË¥°ÁåÆÂõæ

            cam = grad * cam
        cam = cam.clamp(min=0).mean(dim=0)
        cam = cam[0, 1:]
        return cam

    def generate_LRP(self, input, index=None, method="transformer_attribution", is_ablation=False, start_layer=0):
        output = self.model(input)  # Ëé∑ÂæóÊ®°ÂûãÁöÑËæìÂá∫
